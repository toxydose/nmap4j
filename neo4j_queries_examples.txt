//3 most spreaded ports
MATCH (ip:IPAddress)-[:HAS_PORT]->(p:Port)
WITH p.port AS port, p.protocol AS proto, count(DISTINCT ip) AS hostCount
ORDER BY hostCount DESC
LIMIT 3
RETURN port, proto, hostCount

//Top Ports Table
MATCH (ip:IPAddress)-[:HAS_PORT]->(p:Port)
WITH p.port AS port, p.protocol AS proto, count(DISTINCT ip) AS hostCount
ORDER BY hostCount DESC
LIMIT 3

MATCH (n:Network)-[:CONTAINS]->(ip:IPAddress)-[:HAS_PORT]->(p:Port)
WHERE p.port = port AND p.protocol = proto

RETURN 
    n.cidr        AS subnet,
    ip.address    AS ip,
    p.port        AS port,
    p.protocol    AS protocol
ORDER BY subnet, ip, port
